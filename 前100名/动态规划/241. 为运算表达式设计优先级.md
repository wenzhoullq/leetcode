## 题目
给你一个由数字和运算符组成的字符串 expression ，按不同优先级组合数字和运算符，计算并返回所有可能组合的结果。你可以 按任意顺序 返回答案。
## 思路
回溯,但是感觉这种回溯有问题
```java
class Solution {
    List<Integer> ans=new LinkedList<>();
    String expression;
    int cal(int num1,int num2,char c){
        if(c=='+') return num1+num2;
        else if(c=='-') return num1-num2;
        return num1*num2;
    }
    void backtrack(Stack<Integer>s1,Stack<Character>s2,int index){
        if(index==expression.length()){
            while(!s2.isEmpty()){
                int num2=s1.pop(),num1=s1.pop();
                char c=s2.pop();
                s1.push(cal(num1,num2,c));
            }
            ans.add(s1.pop());
            return ;
        }
        while(index<expression.length()){
            int count=0;
            if(Character.isDigit(expression.charAt(index))){
                while(index<expression.length()&&Character.isDigit(expression.charAt(index))){
                    count=count*10+expression.charAt(index)-'0';
                    index++;
                }
                s1.push(count);
            }
            if(s1.size()==1&&index==expression.length()){
                ans.add(s1.pop());
                return ;
            }
            if(s1.size()>=2&&s2.size()>=1){
                int num2=s1.pop(),num1=s1.pop();
                char c=s2.pop();
                s1.push(cal(num1,num2,c));
                Stack<Integer> news1=(Stack<Integer>)s1.clone();
                Stack<Character> news2=(Stack<Character>)s2.clone();
                backtrack(news1,news2,index);
                s1.pop();
                s1.push(num1);
                s1.push(num2);
                s2.push(c);
            }
            if(index<expression.length())s2.push(expression.charAt(index++));
        }
    }
    public List<Integer> diffWaysToCompute(String expression) {
        this.expression=expression;
        Stack<Integer> s1=new Stack<>();
        Stack<Character> s2=new Stack<>();
        backtrack(s1,s2,0);
        return ans;
    }
}
```
