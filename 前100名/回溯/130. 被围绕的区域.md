## 题目
给你一个 m x n 的矩阵 board ，由若干字符 'X' 和 'O' ，找到所有被 'X' 围绕的区域，并将这些区域里所有的 'O' 用 'X' 填充。
## 思路
从边缘为O的进行访问，如果是O的话打标记，最后访问即可
```java
class Solution {
    int[][] direct={{1,0},{-1,0},{0,1},{0,-1}};
    char[][] board;
    int lengthy,lengthx;
    boolean[][] visit;
    void dfs(int y,int x){
        if(y==0||x==0||y==lengthy-1||x==lengthx-1||visit[y][x]||board[y][x]=='X') return ;
        visit[y][x]=true;
        for(int i=0;i<4;i++){
            dfs(y+direct[i][0],x+direct[i][1]);
        }

    }
    public void solve(char[][] board) {
        lengthy=board.length;
        lengthx=board[0].length;
        this.board=board;
        visit=new boolean[lengthy][lengthx];
        for(int i=1;i<lengthx;i++){
            if(board[0][i]=='O') dfs(1,i);
            if(board[lengthy-1][i]=='O') dfs(lengthy-2,i);
        }
        for(int i=1;i<lengthy;i++){
            if(board[i][0]=='O') dfs(i,1);
            if(board[i][lengthx-1]=='O') dfs(i,lengthx-2);
        }
        for(int i=1;i<lengthy-1;i++){
            for(int j=1;j<lengthx-1;j++){
                if(board[i][j]=='O'&&visit[i][j]==false) board[i][j]='X';
            }
        }
    }
}
```
