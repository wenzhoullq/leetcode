## 题目
给定一个 m x n 整数矩阵 matrix ，找出其中 最长递增路径 的长度。

对于每个单元格，你可以往上，下，左，右四个方向移动。 你 不能 在 对角线 方向上移动或移动到 边界外（即不允许环绕）。

## 思路
用普通的回溯会超时，必须枝剪，建立一个记忆化的矩阵，访问过的长度可以直接记录
```java
class Solution {
    int[][] matrix,direct={{1,0},{-1,0},{0,1},{0,-1}},memo;
    int lengthx,lengthy;
    int findpath(int y,int x){
        int ans=1;
        for(int i=0;i<4;i++){
            int tempy=y+direct[i][0],tempx=x+direct[i][1];
            if(tempx>=0&&tempx<lengthx&&tempy>=0&&tempy<lengthy&&matrix[tempy][tempx]>matrix[y][x]){
                if(memo[tempy][tempx]!=0) ans=Math.max(ans,1+memo[tempy][tempx]);
                else {
                    ans=Math.max(ans,1+findpath(tempy,tempx));
                }
            }  
        }
        memo[y][x]=ans;
        return ans;
    }
    public int longestIncreasingPath(int[][] matrix) {
        this.matrix=matrix;
        lengthx=matrix[0].length;lengthy=matrix.length;
        memo=new int[lengthy][lengthx];
        int ans=0;
        for(int i=0;i<lengthy;i++){
            for(int j=0;j<lengthx;j++){
                ans=Math.max(ans,findpath(i,j));
            }
        }
        return ans;
    }
}
```
